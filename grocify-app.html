function doGet(e) {
  var key = (e.parameter && e.parameter.key) ? e.parameter.key : 'default';
  migrateLegacyIfNeeded_(key);
  return ContentService
    .createTextOutput(JSON.stringify(readState_(key)))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var inc = JSON.parse(e.postData.contents || '{}');
  var key = inc.key || 'default';
  migrateLegacyIfNeeded_(key);

  var state = normalizeState_(inc);
  state.version = Date.now();
  writeState_(key, state);

  return ContentService
    .createTextOutput(JSON.stringify({ok:true, version:state.version}))
    .setMimeType(ContentService.MimeType.JSON);
}

function normalizeState_(data) {
  var deleted = unique_((data.deletedIds || []).filter(Boolean));
  var purchases = (data.purchases || []).map(function(p) {
    return {
      id: p.id || Utilities.getUuid(),
      datetime: p.datetime || '',
      store: p.store || '',
      payMode: p.payMode || '',
      total: Number(p.total || 0),
      notes: p.notes || '',
      items: (p.items || []).map(function(it) {
        return {
          id: it.id || Utilities.getUuid(),
          name: it.name || '',
          category: it.category || 'Other',
          qty: String(it.qty || '1'),
          unit: it.unit || 'pcs',
          price: String(it.price || '')
        };
      })
    };
  }).filter(function(p) {
    return p.id && deleted.indexOf(p.id) === -1;
  });

  return {
    purchases: purchases,
    list: (data.list || []).map(function(it) {
      return {
        id: it.id || Utilities.getUuid(),
        name: it.name || '',
        qty: String(it.qty || '1'),
        unit: it.unit || 'pcs',
        est: String(it.est || ''),
        recur: !!it.recur,
        checked: !!it.checked,
        category: it.category || 'Other'
      };
    }),
    budget: Number(data.budget || 0),
    catBudgets: data.catBudgets || {},
    deletedIds: deleted,
    version: Number(data.version || 0)
  };
}

function readState_(key) {
  var sheets = getSheets_(key);
  var purchaseRows = readRows_(sheets.purchases);
  var itemRows = readRows_(sheets.items);
  var listRows = readRows_(sheets.list);
  var settingsRows = readRows_(sheets.settings);
  var deletedRows = readRows_(sheets.deleted);
  var metaRows = readRows_(sheets.meta);

  var byId = {};
  purchaseRows.forEach(function(r) {
    byId[r.id] = {
      id: r.id,
      datetime: r.datetime || '',
      store: r.store || '',
      payMode: r.payMode || '',
      total: Number(r.total || 0),
      notes: r.notes || '',
      items: []
    };
  });

  itemRows.forEach(function(r) {
    if (!byId[r.purchaseId]) return;
    byId[r.purchaseId].items.push({
      id: r.id,
      name: r.name || '',
      category: r.category || 'Other',
      qty: String(r.qty || '1'),
      unit: r.unit || 'pcs',
      price: String(r.price || '')
    });
  });

  var catBudgets = {};
  var budget = 0;
  settingsRows.forEach(function(r) {
    if (r.type === 'budget') budget = Number(r.value || 0);
    if (r.type === 'catBudget' && r.name) {
      catBudgets[r.name] = Number(r.value || 0);
    }
  });

  var version = 0;
  metaRows.forEach(function(r) {
    if (r.key === 'version') version = Number(r.value || 0);
  });

  return {
    purchases: Object.keys(byId).map(function(id) { return byId[id]; }),
    list: listRows.map(function(r) {
      return {
        id: r.id,
        name: r.name || '',
        qty: String(r.qty || '1'),
        unit: r.unit || 'pcs',
        est: String(r.est || ''),
        recur: toBool_(r.recur),
        checked: toBool_(r.checked),
        category: r.category || 'Other'
      };
    }),
    budget: budget,
    catBudgets: catBudgets,
    deletedIds: unique_(deletedRows.map(function(r) { return r.id; }).filter(Boolean)),
    version: version
  };
}

function writeState_(key, state) {
  var sheets = getSheets_(key);

  writeRows_(sheets.purchases,
    ['id','datetime','store','payMode','total','notes'],
    state.purchases.map(function(p) {
      return [p.id, p.datetime, p.store, p.payMode, Number(p.total || 0), p.notes || ''];
    })
  );

  writeRows_(sheets.items,
    ['id','purchaseId','name','category','qty','unit','price'],
    flatten_(state.purchases.map(function(p) {
      return (p.items || []).map(function(it) {
        return [it.id, p.id, it.name || '', it.category || 'Other', String(it.qty || '1'), it.unit || 'pcs', String(it.price || '')];
      });
    }))
  );

  writeRows_(sheets.list,
    ['id','name','qty','unit','est','recur','checked','category'],
    state.list.map(function(it) {
      return [it.id, it.name || '', String(it.qty || '1'), it.unit || 'pcs', String(it.est || ''), !!it.recur, !!it.checked, it.category || 'Other'];
    })
  );

  var settingsRows = [['budget','',Number(state.budget || 0)]];
  Object.keys(state.catBudgets || {}).forEach(function(name) {
    settingsRows.push(['catBudget', name, Number(state.catBudgets[name] || 0)]);
  });
  writeRows_(sheets.settings, ['type','name','value'], settingsRows);

  writeRows_(sheets.deleted, ['id'], unique_(state.deletedIds || []).map(function(id) { return [id]; }));
  writeRows_(sheets.meta, ['key','value'], [['version', Number(state.version || 0)]]);
}

function migrateLegacyIfNeeded_(key) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheets = getSheets_(key);
  if (sheets.meta.getLastRow() > 1 || sheets.purchases.getLastRow() > 1) return;

  var legacyName = legacySheetName_(key);
  var legacy = ss.getSheetByName(legacyName);
  if (!legacy) return;

  var raw = legacy.getRange(1,1).getValue();
  if (!raw) return;

  try {
    var parsed = JSON.parse(raw);
    var state = normalizeState_(parsed);
    state.version = Number(parsed.version || Date.now());
    writeState_(key, state);
  } catch (err) {
    // Fix #13: log migration errors instead of silently swallowing them
    console.error('Grocify: legacy migration failed for key ' + key, err);
  }
}

function getSheets_(key) {
  var base = baseName_(key);
  return {
    purchases: ensureSheet_(base + '_Purchases', ['id','datetime','store','payMode','total','notes']),
    items: ensureSheet_(base + '_Items', ['id','purchaseId','name','category','qty','unit','price']),
    list: ensureSheet_(base + '_List', ['id','name','qty','unit','est','recur','checked','category']),
    settings: ensureSheet_(base + '_Settings', ['type','name','value']),
    deleted: ensureSheet_(base + '_Deleted', ['id']),
    meta: ensureSheet_(base + '_Meta', ['key','value'])
  };
}

function ensureSheet_(name, headers) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(name) || ss.insertSheet(name);
  if (sheet.getLastRow() === 0) {
    sheet.getRange(1,1,1,headers.length).setValues([headers]);
  } else {
    var existing = sheet.getRange(1,1,1,headers.length).getValues()[0];
    if (existing.join('|') !== headers.join('|')) {
      sheet.clearContents();
      sheet.getRange(1,1,1,headers.length).setValues([headers]);
    }
  }
  return sheet;
}

function readRows_(sheet) {
  var lastRow = sheet.getLastRow();
  var lastCol = sheet.getLastColumn();
  if (lastRow < 2 || lastCol === 0) return [];
  var headers = sheet.getRange(1,1,1,lastCol).getValues()[0];
  var values = sheet.getRange(2,1,lastRow-1,lastCol).getValues();
  return values.filter(function(row) {
    return row.join('') !== '';
  }).map(function(row) {
    var out = {};
    headers.forEach(function(h, i) { out[h] = row[i]; });
    return out;
  });
}

function writeRows_(sheet, headers, rows) {
  sheet.clearContents();
  sheet.getRange(1,1,1,headers.length).setValues([headers]);
  if (rows.length) {
    sheet.getRange(2,1,rows.length,headers.length).setValues(rows);
  }
}

function baseName_(key) {
  return ('GT_' + String(key).replace(/[^a-zA-Z0-9_-]/g,'_')).slice(0,30);
}

function legacySheetName_(key) {
  return ('GT_' + String(key).replace(/[^a-zA-Z0-9_-]/g,'_')).slice(0,28);
}

function flatten_(parts) {
  return [].concat.apply([], parts);
}

function unique_(arr) {
  var out = [];
  var seen = {};
  arr.forEach(function(v) {
    if (!v || seen[v]) return;
    seen[v] = true;
    out.push(v);
  });
  return out;
}

function toBool_(value) {
  return value === true || value === 'TRUE' || value === 'true' || value === 1;
}
